00033 /*
00034  @brief The JSON writer class.
00035  
00036  Objective-C types are mapped to JSON types in the following way:
00037  
00038  @li NSNull -> Null
00039  @li NSString -> String
00040  @li NSArray -> Array
00041  @li NSDictionary -> Object
00042  @li NSNumber (-initWithBool:) -> Boolean
00043  @li NSNumber -> Number
00044  
00045  In JSON the keys of an object must be strings. NSDictionary keys need
00046  not be, but attempting to convert an NSDictionary with non-string keys
00047  into JSON will throw an exception.
00048  
00049  NSNumber instances created with the +initWithBool: method are
00050  converted into the JSON boolean "true" and "false" values, and vice
00051  versa. Any other NSNumber instances are converted to a JSON number the
00052  way you would expect.
00053  
00054  */
00055 @interface SBJsonWriter : SBJsonBase {
00056 
00057 @private
00058     BOOL sortKeys, humanReadable;
00059 }

00061 /*
00062  @brief Whether we are generating human-readable (multiline) JSON.
00063  
00064  Set whether or not to generate human-readable JSON. The default is NO, which produces
00065  JSON without any whitespace. (Except inside strings.) If set to YES, generates human-readable
00066  JSON with linebreaks after each array value and dictionary key/value pair, indented two
00067  spaces per nesting level.
00068  */
00069 @property BOOL humanReadable;
00070 
00071 /*
00072  @brief Whether or not to sort the dictionary keys in the output.
00073  
00074  If this is set to YES, the dictionary keys in the JSON output will be in sorted order.
00075  (This is useful if you need to compare two structures, for example.) The default is NO.
00076  */
00077 @property BOOL sortKeys;

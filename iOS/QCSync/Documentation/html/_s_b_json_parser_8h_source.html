00033 /*
00034  @brief The JSON parser class.
00035  
00036  JSON is mapped to Objective-C types in the following way:
00037  
00038  @li Null -> NSNull
00039  @li String -> NSMutableString
00040  @li Array -> NSMutableArray
00041  @li Object -> NSMutableDictionary
00042  @li Boolean -> NSNumber (initialised with -initWithBool:)
00043  @li Number -> (NSNumber | NSDecimalNumber)
00044  
00045  Since Objective-C doesn't have a dedicated class for boolean values, these turns into NSNumber
00046  instances. These are initialised with the -initWithBool: method, and 
00047  round-trip back to JSON properly. (They won't silently suddenly become 0 or 1; they'll be
00048  represented as 'true' and 'false' again.)
00049  
00050  As an optimisation short JSON integers turn into NSNumber instances, while complex ones turn into NSDecimalNumber instances.
00051  We can thus avoid any loss of precision as JSON allows ridiculously large numbers.
00052  
00053  */
00054 @interface QC_SBJsonParser : QC_SBJsonBase {
00055     
00056 @private
00057     const char *c;
00058 }
